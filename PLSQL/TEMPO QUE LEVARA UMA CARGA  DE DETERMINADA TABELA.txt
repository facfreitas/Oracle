PROCEDURE TEMPO_CARGA(DONO IN VARCHAR2 ,NM_TABELA in VARCHAR2)is

CONT NUMBER(9);
TEMPO NUMBER(4);
begin

SELECT COUNT(*) INTO CONT FROM DONO || '.' || NM_TABELA;


SELECT (((758323- CONT)*15)/5000)/60/60  INTO TEMPO 
FROM DONO || '.' || NM_TABELA;

DBMS_OUTPUT.PUT_LINE('A CARGA LEVARÁ ' || TEMPO || ' HORAS');

end;
